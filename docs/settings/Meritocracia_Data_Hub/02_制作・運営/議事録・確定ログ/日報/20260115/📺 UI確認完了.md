#!/usr/bin/env bash

# 色定義
GREEN='\033[0;32m'
RED='\033[0;31m'
CYAN='\033[0;36m'
YELLOW='\033[1;33m'
NC='\033[0m' # No Color

# 累計文字数取得（仮）
TOTAL_CHARS=12450
NEXT_REWARD=13000
PROGRESS=$(( TOTAL_CHARS * 100 / NEXT_REWARD ))

# プログレスバー生成
progress_bar() {
  local percent=$1
  local filled=$(( percent / 5 ))
  local empty=$(( 20 - filled ))
  
  printf "["
  printf "█%.0s" $(seq 1 $filled)
  printf "░%.0s" $(seq 1 $empty)
  printf "]"
}

# メニュー表示
clear
echo "╔════════════════════════════════════════╗"
echo "║  🖋 Merit-Writer CLI (v1.5)           ║"
echo "║  〜アイリスとデイジアの執筆支援〜      ║"
echo "╚════════════════════════════════════════╝"
echo ""
echo "        💜              🦇"
echo "       /|\\            /|\\"
echo "      / | \\          / | \\"
echo "    アイリス        デイジア"
echo ""
echo -e "${GREEN}💜「ご主人様、今日も頑張りましょうね？」${NC}"
echo -e "${RED}🦇「書かねぇと噛むぞ！」${NC}"
echo ""
echo "📊 ステータス"
echo "━━━━━━━━━━━━━━━━━━━━━━━━━━"
echo "累計文字数: ${TOTAL_CHARS} / 次回ご褒美: ${NEXT_REWARD}"
printf "進捗率: "
progress_bar $PROGRESS
printf " ${PROGRESS}%%\n"
echo "━━━━━━━━━━━━━━━━━━━━━━━━━━"
echo ""
echo "───────────────────────────────────────"
echo "1) 📝 新規執筆"
echo "2) ✏️  リライト"
echo "3) 📤 本文抽出＆なろう出力"
echo "4) 🔢 本文文字数（単体）"
echo "5) 🎁 累計文字数（ご褒美あり）"
echo "q) 🚪 終了"
echo "───────────────────────────────────────"
read -rp "選択: " MENU
