🧠 20｜run.sh 設計（完全版）
#!/usr/bin/env bash

PROMPT="$1"
DATE=$(date +"%Y%m%d_%H%M%S")
OUT="output/result_$DATE.txt"
LOG="logs/run.log"

mkdir -p output logs tmp

echo "=== RUN $DATE ===" >> $LOG
echo "PROMPT: $PROMPT" >> $LOG

# AI処理（仮）
echo ">>> AI OUTPUT <<<" > $OUT
echo "$PROMPT" >> $OUT

# Git連携
git add .
git commit -m "auto: $DATE"
git push

echo "DONE: $OUT"
